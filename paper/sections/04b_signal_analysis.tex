\subsection{Signal analysis}
\label{sec:signal-analysis}

\textbf{Purpose.} Test whether each signal layer helps the unsupervised router---i.e., predicts oracle $r(q)$ (appropriate model choice)---independent of routing policy quality.

\textbf{Split.} Calibration set only; compare features to offline $r(q)$ from Equation~\eqref{eq:routing-need}.

\paragraph{Statistics.}
\begin{itemize}
  \item \textbf{Univariate:} Spearman $\rho$, AUROC / AUPRC per feature vs.\ $r(q)$.
  \item \textbf{Layer tests (H1--H3):} AUROC / AUPRC for each layer's feature block alone---$\phi(q)$, $\psi(q,\Mlo)$, $\chi(q)$---vs.\ $r(q)$.
  \item \textbf{Combination models (optional):} joint $\phi{+}\psi$, $\phi{+}\psi{+}\chi$ for selecting frozen $x(q)$ (Section~\ref{sec:signal-selection}); \textbf{not} the definition of H2 (which tests $\psi$ alone) or H3 (which tests $\chi$ alone).
  \item \textbf{Bucket view:} stratified views over $(y_{\mathrm{lo}},y_{\mathrm{hi}})$ buckets (Table~\ref{tab:buckets}).
\end{itemize}

We do \textbf{not} report $\pi(q)$ or accuracy--cost at this stage.
Results appear in Section~\ref{sec:results-signals} (Table~\ref{tab:signal-analysis}).

\paragraph{Hypotheses (motivated by signal assumptions, Section~\ref{sec:signal-extraction}).}
\textbf{H1 (model-independent):} Query-derived signals predict $r(q)$---$\phi(q)$ helps the router (complexity $\to$ appropriate $\Mhi$).
\textbf{H2 (model-dependent):} Model-response signals predict $r(q)$---$\psi(q,\Mlo)$ \emph{alone} helps the unsupervised router (uncertainty $\to$ error).
\textbf{H3 (cross-model):} Cross-model comparative signals predict $r(q)$---$\chi(q)$ \emph{alone} helps the unsupervised router (disagreement $\to$ opportunity).

Each hypothesis is tested on \emph{its own layer's features}, not as incremental gain over the previous layer and not as $\phi{+}\psi$ combination (that supports Stage~7 feature selection).
